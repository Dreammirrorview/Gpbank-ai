function showAdminTab(tab){
document.querySelectorAll('.admin-tab').forEach(t=>t.classList.add('hidden'));
document.getElementById('tab-'+tab).classList.remove('hidden');
document.querySelectorAll('.sidebar-nav .nav-item').forEach(n=>n.classList.remove('active'));
event.target.closest('.nav-item')?.classList.add('active');
document.getElementById('adminPageTitle').textContent={overview:'Dashboard Overview',customers:'Customer Management',accounts:'Accounts & Balance',transfers:'Transfers',payments:'Banks & Payments',crypto:'Crypto Wallet',mining:'Mining Engine',trading:'Trade Dragon',invest:'Investment & Shares',cards:'Virtual Cards',salary:'Salary/Payroll',betting:'Sports Betting',terminal:'Security Terminal',reports:'Reports',settings:'Settings',messages:'Customer Messages'}[tab]||'Dashboard';
}
